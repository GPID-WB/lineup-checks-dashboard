survey_indicator <- reactive({
  # Copy data table
  survey_indicator <- copy(surveys_merged)
  survey_indicator <- survey_indicator[poverty_line == svy_pl(), ]

  if (!svy_chg() == "Ratio") {
    if (svy_chg() == "Percentage change") {
      # indicator
      setnames(
        survey_indicator,
        old = (paste0(svy_ind(), "_perc")),
        new = "New_minus_old"
      )
    } else {
      # indicator
      setnames(
        survey_indicator,
        old = (paste0(svy_ind(), "_diff")),
        new = "New_minus_old"
      )
    }
  } else {
    setnames(
      survey_indicator,
      old = (paste0(svy_ind(), "_ratio")),
      new = "New_minus_old"
    )
  }

  if (!"All" %chin% svy_ct()) {
    survey_indicator <- survey_indicator |>
      fsubset(country_code %in% svy_ct())
  }

  # Tooltip
  survey_indicator[,
    text_tooltip := paste0(
      "Economy: ",
      country_code,
      "\n",
      "Region: ",
      region_name.x,
      "\n",
      "Reporting level: ",
      reporting_level,
      "\n",
      "Value: ",
      round(New_minus_old, 4),
      "\n",
      "Year: ",
      reporting_year
    )
  ]

  survey_indicator
})

survey_baltman <- reactive({
  # Copy data table
  survey_indicator <- surveys_merged
  survey_indicator <- survey_indicator[poverty_line == svy_pl(), ]
  ind <- paste(input$survey_indicator)

  if (!"All" %chin% input$survey_country) {
    survey_indicator <- survey_indicator |>

  survey_indicator[,
    `:=`(
      diff = get(paste0(ind, ".x")) -

      mean_btw = (get(paste0(ind, ".x")) +
        get(paste0(ind, ".y"))) /
        2
    )
  ]
  # Tooltip
  survey_indicator[,
    text_tooltip := paste0(
      "Economy: ",
      country_code,
      "\n",
      "Region: ",
      region_name.x,
      "\n",
      "Reporting level: ",
      reporting_level,
      "\n",
      "Mean old and new: ",
      round(mean_btw, 4),
      "\n",
      "Difference: ",
      round(diff, 4),
      "\n",
      "Year: ",
      reporting_year
    )
  ]
  survey_indicator
})

survey_outlier <- reactive({
  # Copy data table
  survey_indicator <- copy(surveys_merged)
  survey_indicator <- survey_indicator[poverty_line == svy_pl(), ]

  # Define low and high thresholds
  survey_indicator <- survey_indicator |>
    fmutate(
      New = get(paste0(svy_ind(), ".x")),
      Old = get(paste0(svy_ind(), ".y")),
      new_old = abs(New - Old) / Old
    )
  # survey_indicator[, new_old :=
  #                abs(get(svy_ind(), ".x") - get(svy_ind(), ".y")) /
  #                  get(svy_ind(), ".y")]
  survey_indicator[,
    `:=`(
      thres_new_old_lo = mean(new_old, na.rm = TRUE) -
        input$survey_outlier * sd(new_old, na.rm = TRUE),
      thres_new_old_hi = mean(new_old, na.rm = TRUE) +
        input$survey_outlier * sd(new_old, na.rm = TRUE)
    )
  ]

  # Inside or outside threshold
  survey_indicator <-
    if (input$survey_outlier_yes == "Yes") {
      survey_indicator[!between(new_old, thres_new_old_lo, thres_new_old_hi)]
    } else {
      survey_indicator[between(new_old, thres_new_old_lo, thres_new_old_hi)]
    }

  survey_indicator <- survey_indicator |>
    fselect("country_code", "reporting_year", "welfare_type", "Old", "New")

  setnames(
    survey_indicator,
    old = c("Old", "New"),
    new = c(paste0(c("Old_", "New_"), svy_ind()))
  )

  survey_indicator
})
